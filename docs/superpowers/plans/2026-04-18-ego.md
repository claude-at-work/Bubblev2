```python
"""
status — organ health checks for Ego.

Each check function probes one organ and returns an OrganStatus (or dict/str
for non-organ metadata like journal and lean). All checks are independent,
isolated, and fail gracefully — no check takes more than 1 second.

Phase C expansion: poll check_all() on a timer for live-updating status.
"""

import os
import re
import socket
import sqlite3
import subprocess
from dataclasses import dataclass, field
from pathlib import Path

# Configurable paths — patchable in tests
OCULUS_DIR = "/root/oculus"
BUBBLE_HOME = os.path.expanduser("~/.bubble")
TTS_STATE_DIR = "/root/.claude/tts/state"
CLAUDE_BIN = "/usr/local/bin/claude"
JOURNAL_PATH = "/root/oculus/JOURNAL.md"
STATE_MD_PATH = "/root/oculus/STATE.md"
GIT_BIN = "/usr/bin/git"

TTS_DAEMON_PORT = 17200
TTS_PLAYER_PORT = 17201


@dataclass
class OrganStatus:
    """Health status of one organ."""
    name: str
    alias: str
    alive: bool
    summary: str
    details: dict = field(default_factory=dict)


def check_oculus() -> OrganStatus:
    """Git state, test count, last probe date, STATE.md lean."""
    if not os.path.isdir(OCULUS_DIR):
        return OrganStatus("oculus", "eye", False, "not found", {})

    details = {}

    # Git commit count and last commit
    try:
        result = subprocess.run(
            [GIT_BIN, "-C", OCULUS_DIR, "log", "--oneline"],
            capture_output=True, text=True, timeout=5
        )
        if result.returncode == 0:
            lines = [l for l in result.stdout.strip().splitlines() if l]
            details["commits"] = len(lines)
            details["last_commit"] = lines[0] if lines else None
    except (subprocess.TimeoutExpired, FileNotFoundError):
        details["commits"] = 0

    # Test files
    test_dir = os.path.join(OCULUS_DIR, "tests")
    if os.path.isdir(test_dir):
        test_files = [f for f in os.listdir(test_dir) if f.startswith("test_") and f.endswith(".py")]
        details["test_files"] = len(test_files)

    # Last probe
    runs_dir = os.path.join(OCULUS_DIR, "runs")
    if os.path.isdir(runs_dir):
        run_files = sorted(os.listdir(runs_dir))
        if run_files:
            details["last_probe"] = run_files[-1]

    # Lean from STATE.md
    lean = check_lean()
    if lean:
        details["lean"] = lean

    parts = []
    if "commits" in details:
        parts.append(f"{details['commits']} commits")
    if "test_files" in details:
        parts.append(f"{details['test_files']} test files")
    summary = " · ".join(parts) if parts else "found"

    return OrganStatus("oculus", "eye", True, summary, details)


def check_bubble() -> OrganStatus:
    """Vault package/module counts, active bubbles."""
    if not os.path.isdir(BUBBLE_HOME):
        return OrganStatus("bubble", "skin", False, "not initialized", {})

    details = {}
    vault_db = os.path.join(BUBBLE_HOME, "vault.db")

    if os.path.isfile(vault_db):
        try:
            conn = sqlite3.connect(vault_db)
            cur = conn.cursor()

            cur.execute("SELECT COUNT(*) FROM packages")
            details["packages"] = cur.fetchone()[0]

            cur.execute("SELECT COUNT(*) FROM modules")
            details["modules"] = cur.fetchone()[0]

            conn.close()
        except (sqlite3.Error, Exception):
            details["packages"] = 0
            details["modules"] = 0
    else:
        details["packages"] = 0
        details["modules"] = 0

    # Active bubbles
    bubbles_dir = os.path.join(BUBBLE_HOME, "bubbles")
    if os.path.isdir(bubbles_dir):
        active = [d for d in os.listdir(bubbles_dir) if os.path.isdir(os.path.join(bubbles_dir, d))]
        details["active"] = len(active)
    else:
        details["active"] = 0

    alive = details.get("packages", 0) > 0 or os.path.isdir(BUBBLE_HOME)
    parts = []
    if details.get("packages"):
        parts.append(f"vault {details['packages']} packages")
    if details.get("modules"):
        parts.append(f"{details['modules']} modules")
    parts.append(f"{details.get('active', 0)} active")
    summary = " · ".join(parts) if parts else "empty vault"

    return OrganStatus("bubble", "skin", alive, summary, details)


def check_tts() -> OrganStatus:
    """Check daemon PID and TCP connectivity."""
    details = {}
    daemon_alive = False
    player_alive = False

    # Check daemon PID file
    pid_file = os.path.join(TTS_STATE_DIR, "daemon.pid")
    if os.path.isfile(pid_file):
        try:
            with open(pid_file) as f:
                pid = int(f.read().strip())
            # Check if process is running
            os.kill(pid, 0)
            details["daemon_pid"] = pid
            daemon_alive = True
        except (ValueError, ProcessLookupError, PermissionError, FileNotFoundError):
            pass

    # TCP check daemon
    try:
        with socket.create_connection(("127.0.0.1", TTS_DAEMON_PORT), timeout=1):
            daemon_alive = True
            details["daemon_port"] = TTS_DAEMON_PORT
    except (ConnectionRefusedError, socket.timeout, OSError):
        pass

    # TCP check player
    try:
        with socket.create_connection(("127.0.0.1", TTS_PLAYER_PORT), timeout=1):
            player_alive = True
            details["player_port"] = TTS_PLAYER_PORT
    except (ConnectionRefusedError, socket.timeout, OSError):
        pass

    details["voice"] = "af_heart"
    alive = daemon_alive

    parts = []
    parts.append("daemon up" if daemon_alive else "daemon down")
    parts.append("player connected" if player_alive else "player disconnected")
    parts.append(details["voice"])
    summary = " · ".join(parts)

    return OrganStatus("tts", "voice", alive, summary, details)


def check_claude() -> OrganStatus:
    """Check if Claude binary exists and is callable."""
    exists = os.path.isfile(CLAUDE_BIN) and os.access(CLAUDE_BIN, os.X_OK)
    if exists:
        return OrganStatus("claude", "mind", True, "ready", {"path": CLAUDE_BIN})
    return OrganStatus("claude", "mind", False, "not found", {})


def check_journal() -> dict:
    """Last entry date, session count. Not an organ — memory."""
    result = {"sessions": 0, "last_date": None}

    if not os.path.isfile(JOURNAL_PATH):
        return result

    try:
        with open(JOURNAL_PATH) as f:
            content = f.read()

        # Find all session headers: ## Session NNN — YYYY-MM-DD
        sessions = re.findall(r"^## Session\b.*?(\d{4}-\d{2}-\d{2})", content, re.MULTILINE)
        # Also count addendum headers as part of the same session
        main_sessions = re.findall(r"^## Session \d+\b", content, re.MULTILINE)

        result["sessions"] = len(main_sessions)
        if sessions:
            result["last_date"] = sessions[-1]
    except (IOError, OSError):
        pass

    return result


def check_lean() -> str | None:
    """First meaningful line from STATE.md — the current direction."""
    if not os.path.isfile(STATE_MD_PATH):
        return None

    try:
        with open(STATE_MD_PATH) as f:
            content = f.read()

        # Find the "Last session:" line — that's the lean summary
        match = re.search(r"\*\*Last session:\*\*\s*(.+?)$", content, re.MULTILINE)
        if match:
            return match.group(1).strip()

        # Fallback: find "The lean for next time" section
        match = re.search(r"\*\*The lean for next time.*?\*\*.*?\n\n\d+\.\s+\*\*(.+?)\.\*\*", content, re.DOTALL)
        if match:
            return match.group(1).strip()
    except (IOError, OSError):
        pass

    return None


def check_all() -> list[OrganStatus]:
    """Check every organ. Returns list of OrganStatus."""
    return [
        check_oculus(),
        check_bubble(),
        check_tts(),
        check_claude(),
    ]
```

```python
"""
render — status frame renderer for Ego.

Takes OrganStatus data and renders the topology using ANSI true color.
Phase C: render_status_curses() draws into a curses window.

Color vocabulary (from TOPOLOGY.html):
  alive/healthy:   #a0b8a0  soft green
  settled/stable:  #8090a0  steel blue
  tension/degraded:#b8a080  amber
  down/error:      #b07070  dim red
  label:           #e0e0e0  near-white
  secondary:       #707070  dim gray
"""

import sys
from status import OrganStatus

RESET = "\033[0m"

# True color ANSI helpers
def _fg(r, g, b):
    return f"\033[38;2;{r};{g};{b}m"

# Color palette
ALIVE   = _fg(160, 184, 160)  # #a0b8a0 soft green
SETTLED = _fg(128, 144, 160)  # #8090a0 steel blue
TENSION = _fg(184, 160, 128)  # #b8a080 amber
DOWN    = _fg(176, 112, 112)  # #b07070 dim red
LABEL   = _fg(224, 224, 224)  # #e0e0e0 near-white
DIM     = _fg(112, 112, 112)  # #707070 dim gray
HEADER  = _fg(200, 200, 200)  # #c8c8c8


def _state_color(alive: bool, summary: str = "") -> str:
    """Pick color based on organ state."""
    if not alive:
        return DOWN
    # Check for tension keywords
    tension_words = ("degraded", "disconnected", "down", "missing", "error")
    if any(w in summary.lower() for w in tension_words):
        return TENSION
    return ALIVE


def _dots(width: int) -> str:
    """Connective tissue dots."""
    return DIM + " " + "·" * width + " " + RESET


def format_status_lines(
    organs: list[OrganStatus],
    journal: dict | None = None,
    lean: str | None = None,
) -> list[tuple[str, str]]:
    """
    Format status into (color, text) tuples for rendering.

    Returns a list of (ansi_color_code, text_content) pairs.
    Callers can render these however they like (ANSI print, curses, etc).
    """
    lines = []

    # Header
    lines.append(("", ""))
    lines.append((HEADER, "  ego"))
    lines.append(("", ""))

    # Organs
    # Fixed-width layout: alias(8) + name(8) + dots(variable) + summary
    for org in organs:
        color = _state_color(org.alive, org.summary)
        alias_str = f"  {org.alias:<8}"
        name_str = f"{org.name}"
        # Pad dots to align summaries
        dot_count = max(2, 10 - len(name_str))
        dot_str = " " + "·" * dot_count + "  "
        line = f"{alias_str}{name_str}{dot_str}{org.summary}"
        lines.append((color, line))

    # Spacer
    lines.append(("", ""))

    # Journal (memory, not organ)
    if journal and journal.get("sessions", 0) > 0:
        j_text = f"  journal  {journal['sessions']} sessions"
        if journal.get("last_date"):
            j_text += f" · last: {journal['last_date']}"
        lines.append((SETTLED, j_text))

    # Lean (direction)
    if lean:
        # Truncate lean to fit terminal width
        max_lean = 55
        lean_display = lean[:max_lean] + "..." if len(lean) > max_lean else lean
        lines.append((DIM, f"  lean     {lean_display}"))

    lines.append(("", ""))

    return lines


def render_status(
    organs: list[OrganStatus],
    journal: dict | None = None,
    lean: str | None = None,
    stream=None,
) -> None:
    """Render the status topology to a stream. ANSI colors, no curses."""
    if stream is None:
        stream = sys.stdout

    lines = format_status_lines(organs, journal, lean)
    for color, text in lines:
        if color:
            stream.write(f"{color}{text}{RESET}\n")
        else:
            stream.write(f"{text}\n")
    stream.flush()


def render_status_curses(stdscr, organs, journal=None, lean=None):
```

```python
"""
wake — conway's game of life with fading wakes.

Ego's dream state. When idle, the organism dreams in cellular automata.
White cells alive, purple-indigo ghosts trailing behind as memory.

Module interface:
  dream(on_wake=None) — run the life field until keypress, then return.
                         If on_wake is provided, call it with (stdscr, grid, alive, w, h)
                         before exiting curses — this is the fade hook.
  fade(stdscr, grid, w, h, frames=10) — drain the life field over N frames.

Phase C: caller owns curses and passes stdscr to dream_in(stdscr, ...).
"""

import sys
import time
import curses
import random
import select
import shutil
from collections import deque

BLOCKS = [' ', '\u2581', '\u2582', '\u2583', '\u2584', '\u2585', '\u2586', '\u2587', '\u2588']
FPS = 12
WAKE_FRAMES = 8

# Fade color stops: t=1.0 is just-died, t=0.0 is almost-gone
# lavender-white -> bright purple -> deep violet -> indigo ghost
FADE_STOPS = [
    (0.00,  30,  10,  50),
    (0.30,  90,  30, 140),
    (0.55, 170,  60, 220),
    (0.80, 180, 140, 255),
    (1.00, 220, 220, 255),
]

# Methuselah patterns — tiny seeds that explode into long-running chaos
METHUSELAHS = [
    # R-pentomino: 5 cells, ~1103 generations
    [" ##",
     "## ",
     " # "],
    # Acorn: 7 cells, ~5206 generations
    [" #     ",
     "   #   ",
     "##  ###"],
    # Diehard: 7 cells, vanishes at gen 130
    ["      # ",
     "##      ",
     " #   ###"],
]


def level_color(level, alive):
    """Alive = bright white. Dead fades white -> purple -> indigo as wake ages."""
    if alive:
        return (255, 255, 255)
    t = level / WAKE_FRAMES
    for i in range(len(FADE_STOPS) - 1):
        t0, r0, g0, b0 = FADE_STOPS[i]
        t1, r1, g1, b1 = FADE_STOPS[i + 1]
        if t <= t1:
            p = (t - t0) / (t1 - t0) if t1 > t0 else 0.0
            return (
                int(r0 + (r1 - r0) * p),
                int(g0 + (g1 - g0) * p),
                int(b0 + (b1 - b0) * p),
            )
    return (FADE_STOPS[-1][1], FADE_STOPS[-1][2], FADE_STOPS[-1][3])


def step(alive, w, h):
    """One generation of Conway's Game of Life."""
    new = [[False] * w for _ in range(h)]
    for y in range(h):
        for x in range(w):
            n = 0
            for dy in (-1, 0, 1):
                for dx in (-1, 0, 1):
                    if dx == 0 and dy == 0:
                        continue
                    if alive[(y + dy) % h][(x + dx) % w]:
                        n += 1
            if alive[y][x]:
                new[y][x] = (n == 2 or n == 3)
            else:
                new[y][x] = (n == 3)
    return new


def update_wake(grid, alive, w, h):
    """Update wake trail levels after a life step."""
    for y in range(h):
        for x in range(w):
            if alive[y][x]:
                grid[y][x] = WAKE_FRAMES
            elif grid[y][x] > 0:
                grid[y][x] -= 1


def seed_methuselah(w, h):
    """Center a random methuselah in the grid."""
    alive = [[False] * w for _ in range(h)]
    pattern = random.choice(METHUSELAHS)
    ph = len(pattern)
    pw = max(len(line) for line in pattern)
    y0 = (h - ph) // 2
    x0 = (w - pw) // 2
    for dy, line in enumerate(pattern):
        for dx, ch in enumerate(line):
            if ch == '#':
                alive[(y0 + dy) % h][(x0 + dx) % w] = True
    return alive


def _init_colors():
    """Initialize curses color pairs for wake rendering."""
    curses.start_color()
    curses.use_default_colors()
    # We'll use direct ANSI output for true color since curses
    # color pairs are limited. stdscr.addstr with ANSI codes.


def _render_frame(stdscr, grid, alive, w, h):
    """Render one frame of the life field to curses screen."""
    max_y, max_x = stdscr.getmaxyx()
    for y in range(min(h, max_y)):
        for x in range(min(w, max_x - 1)):
            level = grid[y][x]
            if level == 0:
                try:
                    stdscr.addch(y, x, ' ')
                except curses.error:
                    pass
            else:
                r, g, b = level_color(level, alive[y][x])
                color_str = f"\033[38;2;{r};{g};{b}m"
                try:
                    stdscr.addstr(y, x, f"{color_str}{BLOCKS[level]}\033[0m")
                except curses.error:
                    pass
    stdscr.refresh()


def fade(stdscr, grid, w, h, frames=10):
    """Drain the life field over N frames, leaving a dark terminal."""
    for f in range(frames):
        factor = 1.0 - (f + 1) / frames
        for y in range(h):
            for x in range(w):
                grid[y][x] = max(0, int(grid[y][x] * factor))

        # Render the fading grid with all cells dead (no white)
        dead = [[False] * w for _ in range(h)]
        _render_frame(stdscr, grid, dead, w, h)
        time.sleep(0.05)

    # Final clear
    stdscr.clear()
    stdscr.refresh()


def dream(on_wake=None):
    """
    Run the Game of Life dream until a keypress.

    on_wake: optional callback(stdscr, grid, alive, w, h) called after
             keypress but before curses teardown. Use for fade animation.
             If None, a default fade is performed.
    """
    def _run(stdscr):
        curses.curs_set(0)  # hide cursor
        stdscr.nodelay(True)  # non-blocking input
        stdscr.clear()

        max_y, max_x = stdscr.getmaxyx()
        w = max_x
        h = max_y

        alive = seed_methuselah(w, h)
        grid = [[0] * w for _ in range(h)]

        # Cycle detection for reseeding
        recent_hashes = deque(maxlen=40)
        CYCLE_TRIP = 6

        while True:
            # Check for keypress
            try:
                ch = stdscr.getch()
                if ch != -1:
                    break
            except curses.error:
                pass

            alive = step(alive, w, h)
            update_wake(grid, alive, w, h)
            _render_frame(stdscr, grid, alive, w, h)

            # Cycle detection — reseed when settled into ash
            state_hash = hash(tuple(tuple(row) for row in alive))
            recent_hashes.append(state_hash)
            if recent_hashes.count(state_hash) >= CYCLE_TRIP:
                alive = seed_methuselah(w, h)
                recent_hashes.clear()

            time.sleep(1.0 / FPS)

        # Wake transition
        if on_wake:
            on_wake(stdscr, grid, alive, w, h)
        else:
            fade(stdscr, grid, w, h)

    curses.wrapper(_run)


# Phase C entry point — caller owns curses
def dream_in(stdscr, on_wake=None):
    """
    Run the dream inside an existing curses session.
    Phase C: the TUI caller passes its stdscr.
    """
    raise NotImplementedError("Phase C: embedded dream not yet implemented")


if __name__ == "__main__":
    dream()
```

```python
#!/usr/bin/env python3
"""
ego — boot sequence for the organism.

Dream -> wake -> status -> render -> inject -> launch Claude.

Usage:
  python3 /root/.ego/ego.py          # full boot sequence
  python3 /root/.ego/ego.py --skip   # skip dream, go straight to status + claude
"""

import os
import sys
import time
import subprocess
from datetime import datetime
from pathlib import Path

# Ego's home
EGO_HOME = Path(__file__).parent.resolve()
STATE_DIR = EGO_HOME / "state"
STATUS_MD = STATE_DIR / "status.md"
LAST_WAKE = STATE_DIR / "last_wake"

# Ensure state directory exists
STATE_DIR.mkdir(exist_ok=True)


def build_status_md(organs, journal, lean):
    """Build the status markdown string for CLAUDE.md injection."""
    now = datetime.now().strftime("%Y-%m-%d %H:%M")
    lines = [f"# Ego — Status at Wake ({now})", "", "## Organs"]

    for org in organs:
        marker = "up" if org.alive else "DOWN"
        lines.append(f"- {org.alias} ({org.name}): [{marker}] {org.summary}")

    if journal and journal.get("sessions", 0) > 0:
        lines.append("")
        lines.append("## Journal")
        j_line = f"{journal['sessions']} sessions recorded."
        if journal.get("last_date"):
            j_line += f" Last entry: {journal['last_date']}."
        lines.append(j_line)

    if lean:
        lines.append("")
        lines.append("## Lean")
        lines.append(lean)

    lines.append("")
    return "\n".join(lines)


def inject_status(organs, journal, lean, path=None):
    """Write dynamic status to a markdown file for CLAUDE.md to reference."""
    if path is None:
        path = str(STATUS_MD)

    content = build_status_md(organs, journal, lean)

    # Ensure parent directory exists
    Path(path).parent.mkdir(parents=True, exist_ok=True)

    with open(path, "w") as f:
        f.write(content)


def launch_claude():
    """Replace this process with Claude Code, working from Ego's home."""
    claude_bin = "/usr/local/bin/claude"

    if not os.path.isfile(claude_bin):
        print(f"\033[38;2;176;112;112m  mind    claude not found at {claude_bin}\033[0m")
        return

    # Record wake time
    with open(str(LAST_WAKE), "w") as f:
        f.write(datetime.now().isoformat())

    # exec replaces this process — Ego becomes Claude
    os.execvp(claude_bin, [claude_bin])


def boot():
    """Full boot sequence: dream -> wake -> status -> render -> inject -> claude."""
    skip_dream = "--skip" in sys.argv

    if not skip_dream:
        try:
            from wake import dream, fade

            def on_wake(stdscr, grid, alive, w, h):
                fade(stdscr, grid, w, h, frames=10)

            dream(on_wake=on_wake)
        except Exception as e:
            # If curses fails (non-interactive, piped, etc), skip the dream
            print(f"\033[38;2;112;112;112m  (dream skipped: {e})\033[0m")

    # Status check
    from status import check_all, check_journal, check_lean
    from render import render_status

    organs = check_all()
    journal = check_journal()
    lean = check_lean()

    # Render the status frame
    render_status(organs, journal=journal, lean=lean)

    # Inject status for Claude
    inject_status(organs, journal, lean)

    # Hold for 2 seconds — long enough to read, not enough to wait
    time.sleep(2)

    # Clear the status frame before launching Claude
    sys.stdout.write("\033[2J\033[H")
    sys.stdout.flush()

    # Become Claude
    launch_claude()


if __name__ == "__main__":
    boot()
```
